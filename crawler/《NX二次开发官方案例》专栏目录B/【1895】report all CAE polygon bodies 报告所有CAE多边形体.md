    static void report_object_info(char *what, tag_t object)  
    {

                else  
                    ECHO(" (from root occ of)", msg);  
            }  
        }  
        if (!UF_CALL(UF_OBJ_ask_owning_part(object, &owner)))  
        {  
            ask_part_shortname(owner, owning_part);  
            ECHO(" owned by: %s", owning_part);  
        }  
        if (!uc6409(object, &status, name) && (status == 1))  
        {  
            ECHO(", view dependent: %s", name);  
        }  
        if (!UF_OBJ_ask_name(object, name))  
        {  
            ECHO(", name: %s", name);  
        }  
        UF_CALL(UF_SO_is_subclass(object, &is_so));  
        if (is_so)

            }  
            else  
            {  
                ECHO(", visible SO");

#### 代码解析

> 这段代码是一个NX Open C++ API的二次开发示例，其主要功能是遍历模型中的所有多边形体，并打印出每个多边形体的信息，包括类型、子类型、名称、所属组件等信息。
>
> 主要步骤包括：
>
> 1. 初始化NX环境。
> 2. 调用UF_SF_ask_all_polygon_bodies()获取所有多边形体的tag。
> 3. 遍历所有多边形体，并调用report_object_info()打印每个多边形体的详细信息。
> 4. 释放多边形体tag数组。
> 5. 终止NX环境。
>
> 其中，report_object_info()函数用于打印出指定对象的各种信息，包括状态、类型、子类型、名称、所属组件、显示属性等。ask_object_type_descriptor()函数用于根据对象类型和子类型获取描述字符串。ask_part_shortname()函数用于获取组件的短名称。
>
> 该示例代码展示了如何使用NX Open C++ API获取模型信息，并提供了打印对象信息的参考实现。
